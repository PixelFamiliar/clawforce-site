Example: Runtime Gate

Use the SDK to gate skill loading at runtime. Reject untrusted skills before they execute.

// Before loading any skill in your agent framework
const { ClawForce } = require('@clawforce/sdk');
const client = new ClawForce();

async function loadSkill(skillName) {
    const trust = await client.verify(skillName);

    if (!trust.trusted) {
        console.error(`Rejected: ${skillName} (score: ${trust.risk_score})`);
        return null;
    }

    // Safe to load
    return require(skillName);
}

ClawForce scans source code, not framework configs. If your agent runs Python, JavaScript, TypeScript, or shell scripts, it works.
